    , path = this.path

    , tail = '\r\n--' + boundary + '--\r\n'
    , len = tail.length

    buf('\r\n--' + boundary + '\r\n');
    buf('Content-Type: application/atom+xml; charset=UTF-8\r\n');
    buf('\r\n');
    buf(self.metadata());

    len += stat.size;
    buf('\r\n--' + boundary + '\r\n');
    buf('Content-Type: video/webm\r\n');

      , 'Content-Length': len
      , 'Slug': path

    var req = http.request(options, function(res){
      // failed
      if (res.statusCode >= 400) {
